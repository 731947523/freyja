介绍 ​

Freyja-SDK 包含数数、登录、Adjust归因，降低业务项目集成成本，便于业务快速上线

安装使用 ​

md
<script setup>
    import { Freyja } from '@apus/freyja'

    const params = {
        packageName: string // 包名
        versionCode: number | string // 版本号
        versionName: string // 版本名称
        ta: {
            taStatus: boolean // 数数开关
            isNotActive?: boolean // 是否需要激活打点
            appId: string // 数数ID
            serverUrl: string // 数数上报地址
            fn?: (res:any)=>void // 数数加载成功 立即执行的函数
        }
        pay?: {
            ppStatus: boolean // 是否paypal支付
            stStatus: boolean // 是否stripe支付
        }
        login?: { // 是否前置加载登录js 非OAuth2.0形式需前置加载
            ggStatus?: boolean // 谷歌登录-JS加载 (也可业务加载)
            fbStatus?: boolean // Facebook登录-JS加载 (也可业务加载)
        }
    }
    const FreyjaSDK = new Freyja(params)
</script>

常量信息 ​

登录模块 ​

-- 暂不支持FB

javascript

// 登录  OAuth方式
const params: LoginParams = { // 参考ts声明的入参
    type: LOGIN_SOURCE[0] // 默认google登录
    redirectUrl?: string // 回调地址
    scope?: string // 权限
}
FreyjaSDK.oAuthLogin(params)

/* 
*   初始化登录sdk数据 - 仅是初始化，保证后面sdk可用
*/ 
const params: InitLoginInfo = {
    type: LOGIN_SOURCE[0]
    cliendId: string // googleClientId || FB的appid
    sharedDomain?: string //    gg 主域共享 Cookie
    fn?: (val?: any) => void // gg登录成功回调用户信息
    version?: string // FB 使用
}
FreyjaSDK.initLogin(params)

/* 
*   渲染登录按钮
*       注：
*           type为gg时，dom必传
*      params: {
*           type: LOGIN_SOURCE[0], 
*           dom: '', // 必传
*           ggParams: {} //  google按钮样式 默认设置:  type:'standard', outline: 'outline' 
*       }  
*/
const params: {type: string, [key:string]:any}
FreyjaSDK.renderBtn(params)


// 登录 - fb
const params = {
    scope: '' // 授权范围   不传 默认: public_profile
}
FreyjaSDK.loginFb(params)

// 登出 - fb
FreyjaSDK.logoutFb()

// 检测登录状态
const params = {
    type: '' // LOGIN_SOURCE[0]
    [key:string]: any // 拓展保留，咱未支持
}
const fn?: () => {} // 执行时机: 检测登录成功后执行事件(依据业务情形进行使用)
FreyjaSDK.checkLoginStatus(params, fn)

/* 
*   设置登录用户
*       集成式设置Ta用户信息  - 数数登录 + 数数设置用户ID与用户名
*/
const params = {
    supa_no: number // 必传
    nickname: string // 必传
    [key: string]: any
}
FreyjaSDK.setInfo(params)
